			</div>

JavaScript Descriptor

shortcodes.register('hero', {
	anchor: '.shortcode-landing',
	bind_fn: 'append',
	template: 'section.hero',
	elements: {
		img: {
			count: 1,
			bind_fn: 'css',
			bind_property: 'background-image',
			extract_fn: 'attr',
			extract_attr: 'src',
			anchor: '.image-landing',
			anchor_element: 'template'
		},
		a: {
			count: 1,
			anchor: '.link-landing',
			extract_fn: 'attr',
			extract_attr: ['href', 'title', 'html'],
			bind_fn: function(extract, $dest, props) {
				$dest.find(props.anchor).append(''+extract[2]+'');
			},
			anchor_element: 'template'
		},
		h1: {
			count: 1,
			bind_fn: 'html',
			anchor: '.title-landing',
			extract_fn: 'html',
			anchor_element: 'template'
		},
		'h2,h3,h4,h5,h6': {
			count: 1,
			bind_fn: 'html',
			anchor: '.subtitle-landing',
			extract_fn: 'html',
			anchor_element: 'template'
		},
		rest: {
			anchor: '.text-landing',
			bind_fn: 'append',
			extract_fn: 'self',
			anchor_element: 'template'
		}
	}
});

JavaScript Descriptor

shortcodes.register('carousel', {
	anchor: '.shortcode-landing',
	bind_fn: 'append',
	template: 'section.carousel',
	item_template: '.carousel-item',
	item_anchor: '.slides-landing',
	callback: function($template) {
		$template.find('.slides').slick({
			slidesToShow: 3,
			slidesToScroll: 1,
			dots: true,
			arrows: false,
			swipeToSlide: true
		});
	},
	elements: {
		img: {
			count: 'many',
			bind_fn: 'css',
			bind_property: 'background-image',
			extract_fn: 'attr',
			extract_attr: 'src',
			anchor: '.image-landing',
			anchor_element: 'item'
		},
		a: {
			count: 'many',
			anchor: '.link-landing',
			extract_fn: 'attr',
			extract_attr: ['href', 'title', 'html'],
			bind_fn: function(extract, $dest, props) {
				$dest.find(props.anchor).append(''+extract[2]+'');
			},
			anchor_element: 'item'
		},
		h1: {
			count: 'many',
			bind_fn: 'html',
			anchor: '.title-landing',
			extract_fn: 'html',
			anchor_element: 'item'
		},
		rest: {
			anchor: '.text-landing',
			bind_fn: 'append',
			extract_fn: 'self',
			anchor_element: 'item'
		}
	}
});
